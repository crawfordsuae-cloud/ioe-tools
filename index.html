export default { async fetch(request) { if(request.method === "OPTIONS"){ return new Response(null,{ headers:cors() }); } if(request.method !== "POST"){ return json({ status:"API Running" }); } const body = await request.json(); const text = body.text || ""; const address = extract( text, /Address[:\s]+(.+)/i ); const totalFloorArea = extract( text, /total floor area[:\s]+([0-9.]+)/i ); const heatLossPerimeter = extract( text, /heat loss perimeter[:\s]+([0-9.]+)/i ); const roomHeight = extract( text, /room height[:\s]+([0-9.]+)/i ); const roofArea = extract( text, /roof area[:\s]+([0-9.]+)/i ); const windowArea = extract( text, /window area[:\s]+([0-9.]+)/i ); const partyWallPerimeter = extract( text, /party wall perimeter[:\s]+([0-9.]+)/i ); const flatRoofArea = extract( text, /flat roof area[:\s]+([0-9.]+)/i ); const doors = extract( text, /number of doors[:\s]+([0-9.]+)/i ); const currentFuelCost = extract( text, /current fuel cost[:\s£]+([0-9.]+)/i ); const currentCarbon = extract( text, /current carbon[:\s]+([0-9.]+)/i ); const currentRating = extract( text, /current rating[:\s]+([A-Z])/i ); const heatDemand = extract( text, /space heating per m2[:\s]+([0-9.]+)/i ); const loftCost = Number(totalFloorArea || 0) * 12.5; const cavityCost = Number(heatLossPerimeter || 0) * Number(roomHeight || 0) * 11.5; const solarCost = 3250; return json({ address, totalFloorArea, heatLossPerimeter, roomHeight, roofArea, windowArea, partyWallPerimeter, flatRoofArea, doors, currentFuelCost, currentCarbon, currentRating, heatDemand, tenure: "Social Housing", listedBuilding: "No", worldHeritage: "No", measures:[ { name: "Loft Insulation", cost: loftCost.toFixed(2), risk: "Risk Minor", additional:[ "PIBI", "Damp Survey" ] }, { name: "Cavity Wall Insulation", cost: cavityCost.toFixed(2), risk: "Risk Minor", additional:[ "PIBI", "BBA Cavity Wall Survey" ] }, { name: "Solar Photovoltaics", cost: solarCost.toFixed(2), risk: "Risk None", additional:[ "PIBI", "MCS" ] } ] }); } }; function extract(text, regex){ const match = text.match(regex); return match ? match[1] : ""; } function json(data){ return new Response( JSON.stringify(data), { headers:{ "Content-Type": "application/json", ...cors() } } ); } function cors(){ return { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET,POST,OPTIONS", "Access-Control-Allow-Headers": "Content-Type" }; }
